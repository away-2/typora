            const view_activeClass = "active";
            const screenWidth = window.innerWidth || document.documentElement.clientWidth || document.body.clientWidth;

            view_buttons.forEach((button, index) => {
                button.addEventListener("click", () => {
                    view_buttons.forEach((btn) => btn.classList.remove(view_activeClass));
                    view_pages.forEach((page) => (page.style.display = "none"));

                    button.classList.add(view_activeClass);
                    view_pages[index].style.display = "block";
                    if (screenWidth > 848) {
                        createModalData(index);
                    }
                });
            });

            const swiperModal = document.querySelector("#swiper-modal .swiper-wrapper")
            let swiperInstance = null;

            function createModalData(index) {
                swiperModal.innerHTML = "";

                let current_modal_data = document.querySelectorAll(
                    `.static-view-content .view-modal-img${index + 1}`
                );

                let srcArray = Array.from(current_modal_data).map(img => img.src);

                srcArray.forEach(item => {
                    swiperModal.innerHTML += `
                                <div class="swiper-slide">
                                    <img width="1000" height="750" src="${item}" alt="">
                                </div>
                            `;
                });

                current_modal_data.forEach((trigger, i) => {
                    trigger.addEventListener("click", () => {
                        modal.style.display = "block";
                        window.addEventListener('wheel', wheelHandle, { passive: false });

                        if (swiperInstance) {
                            swiperInstance.destroy(true, true);
                        }

                        // 初始化
                        swiperInstance = new Swiper("#swiper-modal", {
                            slidesPerView: 1,
                            allowTouchMove: true,
                            initialSlide: i,
                            navigation: {
                                nextEl: ".swiper-modal-next",
                                prevEl: ".swiper-modal-pre",
                            }
                        });
                    });
                });
            }

            if (screenWidth > 848) {
                createModalData(0);
            }

            const modal = document.querySelector(".view-shadow")
            const closeBtn = document.querySelector(".view-modal-close")
            const modalContent = modal.querySelector(".view-modal-content")
            const swiper_btn = document.querySelectorAll(".swiper-modal-pre, .swiper-modal-next")

            function wheelHandle(e) {
                e.preventDefault()
            }

            closeBtn.addEventListener("click", closeModal)

            // 关闭弹窗
            function closeModal() {
                modal.style.display = "none"
                // document.body.style.overflow = "visible"
                window.removeEventListener('wheel', wheelHandle)
            }

            function setupViewMore(btnSelector, contentSelector) {
                const btn = document.querySelector(btnSelector);
                const contents = document.querySelectorAll(contentSelector);
                let isShow = false;
                if (!btn) return;

                btn.addEventListener("click", () => {
                    contents.style.height = "auto";
                    isShow = !isShow;
                    btn.innerHTML = isShow ? "View Less" : "View More";
                });
            }
