    //自动播放和暂停视频
    function onPlayerReady(event) {
      player.mute(); // 静音

      const observer = new IntersectionObserver((entries) => {
        entries.forEach(entry => {
          if (entry.isIntersecting) {
            player.playVideo();
          } else {
            player.pauseVideo();
          }
        });
      }, {
        threshold: 0.5
      });

      const iframeContainer = document.querySelector('.description-page-container-item-iframe-box');
      if (iframeContainer) {
        observer.observe(iframeContainer);
      } else {
        console.error("Container not found for IntersectionObserver.");
      }
    }
